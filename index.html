How It Works

const capitalizeFirst = (string) => {
  return string.charAt(0).toUpperCase() + string.slice(1);
};

const getRandomInt = (min, max) => {
  return Math.floor(Math.random() * (max - min)) + min; };

const generateName = () => {
  const name1 = ["andrew","niall", "joe", "james", "alan"];
  const name2 = ["hale","barber", "james", "oxley-shaw", "weller"];
  const name = capitalizeFirst(name1[getRandomInt(0, name1.length)])
 + ' '
 + capitalizeFirst(name2[getRandomInt(0, name2.length)]);
  return name;
};
